void setup() {
    pinMode(13, OUTPUT); // Перевод 13-го порта в состояние вывода информации
}

void loop() {
    for(int i = 500; i >= 50; i -= 50) { // Уменьшаем задержку от 500 до 50 миллисекунд с шагом 50
        digitalWrite(13, HIGH); // Включаем светодиод
        delay(i);               // Задержка включения
        digitalWrite(13, LOW);  // Выключаем светодиод
        delay(i);               // Задержка выключения
    }

    // После завершения цикла светодиод остается выключенным на 10 секунд
    delay(10000);
}
